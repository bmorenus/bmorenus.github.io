182// Vector Projection
183inline Vec2D Project(const Vec2D& a, const Vec2D& b) {
184 assert(a.w == 0 && b.w == 0);
185
186 Vec2D vec;
187 float scalarProj = Dot(a, b) / Magnitude(b);
188 Vec2D uB = Normalize(b);
189
190 vec.x = uB.x * scalarProj;
191 vec.y = uB.y * scalarProj;
